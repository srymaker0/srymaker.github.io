class MyQueue {
public:
    /** Initialize your data structure here. */
    stack<int> s1, s2;
    MyQueue() {

    }
    
    /** Push element x to the back of queue. */
    void push(int x) {
        s1.push(x);
        return ;
    }
    
    /** Removes the element from in front of queue and returns that element. */
    int pop() {
        if (s2.empty() == 1) {
            while (!s1.empty()) {
                s2.push(s1.top());
                s1.pop();
            }
        }
        int ret = s2.top();
        s2.pop();
        return ret;     
    }
    
    /** Get the front element. */
    int peek() {
        if (s2.empty() == 1) {
            while (!s1.empty()) {
                s2.push(s1.top());
                s1.pop();
            }
        } 
        return s2.top();    
    }
    
    /** Returns whether the queue is empty. */
    bool empty() {
        return s1.empty() && s2.empty();
    }
};

class Solution {
public:
    string minRemoveToMakeValid(string s) {
        string ret1;
        string ret2;
        int cnt1 = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] != '(' && s[i] != ')') {
                ret1 += s[i];
            } else {
                if (s[i] == '(')  {
                    cnt1++;        
                    ret1 += s[i];
                } else {
                    if (cnt1 > 0) {
                        cnt1--;       
                        ret1 += s[i];              
                    }  
                }
            }
        }
        int cnt2 = 0;
        for (int i = ret1.size() - 1; i >= 0; i--) {
            if (ret1[i] != '(' && ret1[i] != ')') {
                ret2 += ret1[i];
            } else {
                if (ret1[i] == ')')  {
                    cnt2++;        
                    ret2 += ret1[i];
                } else {
                    if (cnt2 > 0) {
                        cnt2--;       
                        ret2 += ret1[i];                
                    }
                }
            }
        }
        reverse(ret2.begin(), ret2.end());
        return ret2;
    }
};
